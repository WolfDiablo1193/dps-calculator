<script>
function calculateDPS() {
    function val(id) {
        return Number(document.getElementById(id).value) || 0;
    }

    // Base and flat DMG
    const totalBaseDmg = val("totalBaseDmg");
    const itemBaseIncreasePct = val("itemBaseIncreasePct");
    const gloveGemRanks = val("gloveGemRanks");
    const paragonDmg = val("paragonDmg");
    const horadrimDmg = val("horadrimDmg");
    const intStrDmg = val("intStrDmg");
    const tourmalineDmg = val("tourmalineDmg");
    const resonance = val("resonance");

    // Modifiers
    const b1Pct = val("b1Pct");
    const b2Pct = val("b2Pct");
    const b3Pct = val("b3Pct");
    const b4Pct = val("b4Pct");
    const critChancePct = val("critChancePct");
    const critDmgPct = val("critDmgPct");
    const attackSpeedPct = val("attackSpeedPct");

    // --- Simplified assumptions (you can adjust later) ---

    // 1) Base DMG after item base increase
    const baseFromItems = totalBaseDmg * (1 + itemBaseIncreasePct / 100);

    // 2) Flat DMG stacking
    const flatExtraDmg = paragonDmg + horadrimDmg + intStrDmg + tourmalineDmg;

    // 3) Simple resonance scaling (tune this later if needed)
    const resonanceMultiplier = 1 + (resonance / 10000); // e.g. 1000 res = +10%

    const flatBaseDmg = (baseFromItems + flatExtraDmg) * resonanceMultiplier;

    // 4) Bracket multipliers
    const b1Mult = 1 + b1Pct / 100;
    const b2Mult = 1 + b2Pct / 100;
    const b3Mult = 1 + b3Pct / 100;
    const b4Mult = 1 + b4Pct / 100;

    // 5) Crit multiplier (expected value):
    //    EV = (1 - critChance) * 1 + critChance * (1 + critDmg)
    const critChance = critChancePct / 100;
    const critBonus = critDmgPct / 100; // 100% -> 1.0
    const critMult = (1 - critChance) * 1 + critChance * (1 + critBonus);

    // 6) Attack speed multiplier
    const asMult = 1 + attackSpeedPct / 100;

    // 7) Final DPS index
    const avgDps = flatBaseDmg * b1Mult * b2Mult * b3Mult * b4Mult * critMult * asMult;

    // Round for display
    const avgDpsRounded = Math.round(avgDps);

    // Breakdown text
    const detailText =
        "Base (items + flat) after resonance: " + flatBaseDmg.toFixed(1) +
        " | B1: x" + b1Mult.toFixed(2) +
        " | B2: x" + b2Mult.toFixed(2) +
        " | B3: x" + b3Mult.toFixed(2) +
        " | B4: x" + b4Mult.toFixed(2) +
        " | Crit: x" + critMult.toFixed(2) +
        " | AS: x" + asMult.toFixed(2);

    document.getElementById("avgDpsResult").innerText = "Avg DPS Score: " + avgDpsRounded.toLocaleString();
    document.getElementById("detailResult").innerText = detailText;
}
</script>
